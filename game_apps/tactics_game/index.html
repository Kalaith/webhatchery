Plugin Integration Guide

Basic Integration

// Initialize the tactical combat plugin
const combat = new TacticalCombatPlugin({
    canvas: document.getElementById('gameCanvas'),
    units: unitData,
    onGameEnd: (winner) => {
        console.log('Game ended:', winner);
    }
});

// Start the tactical combat
combat.initialize();
